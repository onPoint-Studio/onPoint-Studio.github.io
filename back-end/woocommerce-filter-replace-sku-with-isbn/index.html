function translate_woocommerce( $translation, $text, $domain ) {
    if ($domain == 'woocommerce') {
        switch ($text) {
            case 'SKU':
                $translation = 'ISBN';
                break;
            case 'SKU:':
                $translation = 'ISBN:';
                break;
        }
    }
    return $translation;
}

add_filter('gettext', 'translate_woocommerce', 10, 3);
